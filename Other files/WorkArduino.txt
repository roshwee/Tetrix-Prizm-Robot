#include <PRIZM.h>      

PRIZM prizm;           

void setup() 
{
   prizm.PrizmBegin(); 
   prizm.setServoSpeed(1,50);  
   Serial.begin(9600);
   while(!Serial) 
   {
     ;
   }
}

void loop() 
{
   if (Serial.available() > 0) 
   {
      byte incomingByte = Serial.read();
       
      if (incomingByte != -1) 
      {
          if (incomingByte > 0 && incomingByte < 180)
          {
             prizm.setRedLED(HIGH);
             prizm.setServoPosition(1,incomingByte);  
             delay(3000); 
             prizm.setRedLED(LOW);
             delay(3000); 
          }
                                               
        Serial.print("I received: ");
        Serial.print(incomingByte);
      }
   }
}
